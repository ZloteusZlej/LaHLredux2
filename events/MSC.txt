country_event = { #WESSEX SENDS DIVISION#
	id = MSC_event.5
	title = MSC_event.5.t
	desc = MSC_event.5.d
    is_triggered_only = yes
	option = { 
		name = MSC_event.5.a
		MSC = {
			add_equipment_to_stockpile = {
				type = infantry_equipment
				amount = 610
			}
			add_equipment_to_stockpile = {
				type = support_equipment
				amount = 30
			}
			add_equipment_to_stockpile = {
				type = artillery_equipment
				amount = 36
			}
			if = {
				MSC = {
					NOT = {
						has_template = "Wessex Volunteers"
					}	
					division_template = {
						name = "Wessex Volunteers"
						division_names_group = WES_INF_01

						regiments = {
							infantry = { x = 0 y = 0 }	
							infantry = { x = 0 y = 1 }
							infantry = { x = 0 y = 2 }	
							infantry = { x = 0 y = 3 }
							infantry = { x = 0 y = 4 }
							infantry = { x = 0 y = 5 }
							artillery_brigade = { x = 1 y = 0 }
						}
							
						support = { 
							engineer = { x = 0 y = 0 } 	
						}
					}
				}
			}			
			random_owned_controlled_state = {
				create_unit = {
					division = "name = \"Wessex Volunteers\" division_template = \"Wessex Volunteers\" start_experience_factor = 1 start_equipment_factor = 1"
					owner = MSC
				}
			}
		}
    }
}
